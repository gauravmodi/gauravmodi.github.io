Styling tables in Notebook¶

Different color based on value in Column

In [ ]:
def over_10M(value):
    if value > 10**7:
        color = 'green'
    else:
        color = 'blue'        
    return 'color: %s' % color

df.style.applymap(over_10M, subset=['AmountInUSD'])\
        .format({'AmountInUSD': "$ {:}"})
